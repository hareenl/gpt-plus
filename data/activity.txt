import numpy as np
import pymc3 as pm

# Generate some data
np.random.seed(123)
X = np.random.randn(100)
true_slope = 2
true_intercept = 1
sigma = 0.5
y = true_intercept + true_slope*X + np.random.normal(0, sigma, size=100)

# Define the model
with pm.Model() as model:
    slope = pm.Normal('slope', mu=0, sd=10)
    intercept = pm.Normal('intercept', mu=0, sd=10)
    sigma = pm.HalfNormal('sigma', sd=1)

    y_est = slope*X + intercept
    likelihood = pm.Normal('y', mu=y_est, sd=sigma, observed=y)

    # Inference!
    trace = pm.sample(2000, tune=1000, chains=2)
    
# Summarize the trace
pm.summary(trace)
